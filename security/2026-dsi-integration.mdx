```bash

```

### Step 4: Enable GenAI guardrails

```python
# In your API service
from governance_kernel.genai_guardrails import GenAIGuardrail

guardrail = GenAIGuardrail(jurisdiction="KDPA_KE")

# Intercept all GenAI requests
@app.before_request
def check_genai_request():
    if request.path.startswith('/genai'):
        result = guardrail.intercept_request(
            prompt=request.json.get('prompt'),
            endpoint=request.json.get('endpoint'),
            user_id=request.headers.get('User-ID')
        )
        
        if result['action'] == 'block':
            return jsonify({"error": result['reason']}), 403
```
