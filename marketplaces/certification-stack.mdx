### OAuth 2.0 integration

```python
from salesforce.oauth2_integration import SalesforceOAuth2Client

client = SalesforceOAuth2Client(
    client_id=os.getenv("SALESFORCE_CLIENT_ID"),
    client_secret=os.getenv("SALESFORCE_CLIENT_SECRET"),
    redirect_uri="https://iluminara.health/oauth/callback"
)

# Get authorization URL with PKCE
auth_url, code_verifier, state = client.get_authorization_url(
    scope="api refresh_token full"
)
```

### Shield encryption

```python
from salesforce.oauth2_integration import SalesforceShieldEncryption

shield = SalesforceShieldEncryption(client)

# Create encrypted health record
record = shield.create_encrypted_record(
    sobject="Health_Record__c",
    data={
        "Patient_ID__c": "PAT_12345",
        "Diagnosis__c": "Malaria",
        "Symptoms__c": "Fever, Chills"
    },
    sensitive_fields=["Patient_ID__c", "Diagnosis__c"]
)
```

## Microsoft Marketplace

### Azure Purview integration

Automated data classification in `azure/purview_integration.py`:

```python
from azure.purview_integration import PurviewDataClassifier

classifier = PurviewDataClassifier(
    purview_account_name="iluminara-purview",
    subscription_id=os.getenv("AZURE_SUBSCRIPTION_ID"),
    resource_group="iluminara-rg"
)

# Classify health record
classification = classifier.classify_health_record(health_record)

# Register in Purview catalog
asset_guid = classifier.register_asset(
    asset_name="health_record_12345",
    asset_type="health_record",
    classification=classification
)
```
